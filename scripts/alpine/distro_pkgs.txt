pkg_repos_sources() {
	sep='#--------------------#'
	argX='/etc/apk/repositories'
	
	printf "${sep}\n$argX\n" | cat - $argX
	#printf "${sep}\n$argX\n" ; $argX
}

pkgs_installed() {
	METHOD=${1:-explicit}
	
	sleep 3 ; apk update
	
	# apk info for installed # -qr for reverse depns
	pkg_nms=$(apk info)
	(for pkg_nm in $pkg_nms ; do
		if [ -z "$(apk info -qr $pkg_nm)" ] ; then
			echo $pkg_nm ;
		fi
	done) | sort | column
}
